def find_ebitda_definition(definitions: List[Definition]) -> Optional[Definition]:
    """
    Identify the EBITDA-style definition, preferring "Consolidated EBITDA".

    Heuristic scoring:
    - +4 if term contains "consolidated ebitda"
    - +2 if term contains "ebitda" and "consolidated" (any order)
    - +1 if term contains "adjusted"
    - +1 if the definition is relatively long (more detailed)
    - baseline: any term containing "ebitda"

    Parameters
    ----------
    definitions : List[Definition]
        All parsed definitions.

    Returns
    -------
    Optional[Definition]
        The highest-scoring EBITDA definition, if any.
    """
    candidates: List[tuple[int, Definition]] = []

    for d in definitions:
        term_lower = d.term.lower()
        if "ebitda" not in term_lower:
            continue

        score = 1  # baseline for any EBITDA-like term

        if "consolidated ebitda" in term_lower:
            score += 4
        elif "ebitda" in term_lower and "consolidated" in term_lower:
            score += 2

        if "adjusted" in term_lower:
            score += 1

        # Slight preference for longer / more detailed definitions
        if len(d.definition) > 200:
            score += 1

        candidates.append((score, d))

    if not candidates:
        return None

    # Pick the highest-score; tie-breaker = longer definition
    candidates.sort(key=lambda x: (x[0], len(x[1].definition)), reverse=True)
    return candidates[0][1]

def extract_kcps_from_pdf(pdf_path: str) -> ExtractionResult:
    """
    End-to-end extraction pipeline for key credit provisions.

    Steps
    -----
    1. Extract raw text from the PDF.
    2. Parse the text into structured sections.
    3. Extract defined terms from the definitions article(s).
    4. Identify and surface the EBITDA definition.
    5. Locate key covenant sections:
       - Indebtedness (debt baskets)
       - Restricted Payments (RP baskets)
       - Affirmative Covenants
    6. Within those sections, extract clause-level baskets using regex.

    Parameters
    ----------
    pdf_path : str
        Path to the credit agreement PDF.

    Returns
    -------
    ExtractionResult
        Container with all extracted information.
    """
    raw_text = extract_text_from_pdf(pdf_path)
    sections = parse_sections(raw_text)

    # Definitions
    definitions = extract_all_definitions(sections)
    ebitda_def = find_ebitda_definition(definitions)

    # Indebtedness
    indebtedness_sec: Optional[Section] = find_indebtedness_section(sections)
    indebtedness_baskets = (
        extract_baskets_from_section(indebtedness_sec)
        if indebtedness_sec is not None
        else []
    )

    # Restricted Payments
    rp_sec: Optional[Section] = find_restricted_payments_section(sections)
    rp_baskets = (
        extract_baskets_from_section(rp_sec)
        if rp_sec is not None
        else []
    )

    # Affirmative Covenants
    aff_sec: Optional[Section] = find_affirmative_covenants_section(sections)
    aff_baskets = (
        extract_baskets_from_section(aff_sec)
        if aff_sec is not None
        else []
    )

    result = ExtractionResult(
        sections=sections,
        definitions=definitions,
        ebitda_definition=ebitda_def,
        indebtedness_baskets=indebtedness_baskets,
        restricted_payment_baskets=rp_baskets,
        affirmative_covenant_clauses=aff_baskets,
        metadata={"source_pdf": pdf_path},
    )
    return result

def _first_sentence(text: str, max_len: int = 220) -> str:
    """
    Extract a short, readable snippet from a longer legal clause.

    Strategy:
    - Collapse whitespace.
    - Take up to the first period. If that is too long, cut at max_len.
    - Append "..." if truncated.

    Parameters
    ----------
    text : str
        Full clause text.
    max_len : int
        Maximum length of the returned snippet.

    Returns
    -------
    str
        Concise snippet of the clause.
    """
    flat = " ".join(text.replace("\n", " ").split())
    # Try up to first sentence
    m = re.search(r"\.(\s|$)", flat)
    if m:
        candidate = flat[: m.end()].strip()
    else:
        candidate = flat

    if len(candidate) > max_len:
        candidate = candidate[:max_len].rstrip() + " ..."
    return candidate


def print_ebitda_definition(defn: Definition | None) -> None:
    """
    Pretty-print the EBITDA definition, if available.
    """
    print("=" * 80)
    print("EBITDA DEFINITION")
    print("=" * 80)

    if defn is None:
        print("No EBITDA-style definition found.\n")
        return

    print(f"Term: {defn.term}")
    if defn.location_hint:
        print(f"Location: {defn.location_hint}")
    print("\nDefinition (short snippet):")
    snippet = _first_sentence(defn.definition, max_len=400)
    print(indent(snippet, "  "))
    print()


def print_basket_list(title: str, baskets: list[BasketInfo]) -> None:
    """
    Pretty-print a list of BasketInfo records under a section heading.

    Rather than dumping full legal text, show:
    - Section and clause label
    - A short natural-language snippet (first sentence or truncated)
    - Key numeric features (dollar amounts, percentages, ratios, greater-of)
    """
    print("=" * 80)
    print(title)
    print("=" * 80)

    if not baskets:
        print("No items extracted for this category.\n")
        return

    for idx, b in enumerate(baskets, start=1):
        print(f"[{idx}] Section: {b.section_id} - {b.section_title}")
        if b.clause_label:
            print(f"    Clause: {b.clause_label}")

        snippet = _first_sentence(b.raw_text, max_len=260)
        print("    Snippet:")
        print(indent(snippet, "      "))

        # Structured numeric highlights
        if b.dollar_amounts:
            print(f"    Dollar amounts: {', '.join(b.dollar_amounts)}")
        if b.percentages:
            print(f"    Percentages: {', '.join(b.percentages)}")
        if b.ratios:
            print(f"    Ratios: {', '.join(b.ratios)}")
        if b.greater_of_expressions:
            printable = [f"(X: {x}; Y: {y})" for x, y in b.greater_of_expressions]
            print(f"    Greater-of expressions: {', '.join(printable)}")

        print()

"""
Extraction of facility commitments (sizes) and core pricing terms
(Applicable Margin / Rates, floors, commitment fees) using only
regex + simple heuristics.

Strategy
--------
1. Commitments:
   - Look at definitions containing "commitment".
   - Extract the first dollar amount in those definitions.
   - Also scan sections whose titles mention "Commitments" or "Amount";
     extract patterns like "$1,250,000,000 Term Loan B Facility".

2. Pricing:
   - Scan sections whose titles or text contain "Applicable Margin",
     "Applicable Rate", "Interest", or "Fees".
   - Pull percentage values near phrases like "Applicable Margin",
     "Applicable Rate", "commitment fee", "floor".
"""

from __future__ import annotations

import re
from typing import List

from models import (
    Section,
    Definition,
    FacilityCommitment,
    PricingTerm,
)

# Reuse the same dollar / percentage regex styles as covenant_extractor
DOLLAR_RE = re.compile(
    r"\$[\s]*[\d,]+(?:\.\d+)?(?:\s*(?:million|billion|thousand))?",
    re.IGNORECASE,
)

PCT_RE = re.compile(
    r"\b\d+(?:\.\d+)?\s*%",
    re.IGNORECASE,
)


# ---------------------- small helper: snippet maker ----------------------

def _snippet(text: str, max_len: int = 220) -> str:
    flat = " ".join(text.replace("\n", " ").split())
    if len(flat) > max_len:
        return flat[:max_len].rstrip() + " ..."
    return flat


# ---------------------- facility commitments -----------------------------

def extract_facility_commitments_from_definitions(
    definitions: List[Definition],
) -> List[FacilityCommitment]:
    """
    Use definitions containing "commitment" to infer facility sizes.

    Example:
        "Revolving Credit Commitments" means, at any time, the aggregate
        principal amount of $500,000,000 ...

    We grab the first dollar amount and use the term as the facility name.
    """
    results: List[FacilityCommitment] = []

    for d in definitions:
        if "commitment" not in d.term.lower():
            continue

        m = DOLLAR_RE.search(d.definition)
        amount = m.group(0).strip() if m else None
        snippet = _snippet(d.definition)

        results.append(
            FacilityCommitment(
                name=d.term,
                amount_raw=amount,
                context_snippet=snippet,
                location_hint=d.location_hint,
            )
        )

    return results


COMMITMENT_SECTION_KEYWORDS = [
    "commitments",
    "commitment",
    "aggregate principal amount",
    "amount of the term loans",
]


def extract_facility_commitments_from_sections(
    sections: List[Section],
) -> List[FacilityCommitment]:
    """
    Look for obvious facility sizes directly in the body text of sections
    whose titles mention "Commitments" or whose text mentions phrases
    like "aggregate principal amount of $... Term Loan".
    """
    results: List[FacilityCommitment] = []

    FACILITY_REGEX = re.compile(
        r"(aggregate\s+principal\s+amount\s+of\s+"
        r"(?P<amount>\$[\d,\.\s]+(?:million|billion|thousand)?)"
        r"\s+"
        r"(?P<name>(?:[A-Za-z]+\s+){1,5}(?:facility|facilities|term loans|revolving credit commitments)))",
        re.IGNORECASE,
    )

    for sec in sections:
        text_lower = sec.text.lower()
        if not any(k in text_lower or k in sec.title.lower() for k in COMMITMENT_SECTION_KEYWORDS):
            continue

        for match in FACILITY_REGEX.finditer(sec.text):
            amount = match.group("amount").strip()
            name = match.group("name").strip()
            context = _snippet(match.group(0))

            results.append(
                FacilityCommitment(
                    name=name,
                    amount_raw=amount,
                    context_snippet=context,
                    location_hint=sec.id,
                )
            )

    return results


def extract_facility_commitments(
    sections: List[Section],
    definitions: List[Definition],
) -> List[FacilityCommitment]:
    """
    Combine definition-based and section-based commitment detection,
    de-duplicating by (name, amount_raw).
    """
    from_defs = extract_facility_commitments_from_definitions(definitions)
    from_secs = extract_facility_commitments_from_sections(sections)

    combined: List[FacilityCommitment] = []
    seen = set()

    for fc in from_defs + from_secs:
        key = (fc.name.lower(), (fc.amount_raw or "").lower())
        if key in seen:
            continue
        seen.add(key)
        combined.append(fc)

    return combined


# --------------------------- pricing terms -------------------------------

PRICING_SECTION_KEYWORDS = [
    "applicable margin",
    "applicable rate",
    "interest rate",
    "interest; fees",
    "commitment fee",
    "unused fee",
    "default interest",
]


BENCHMARK_HINT_RE = re.compile(
    r"(term\s+sofr|sofr|libor|eurodollar|base rate|alternate base rate|prime rate)",
    re.IGNORECASE,
)


def extract_pricing_terms(sections: List[Section]) -> List[PricingTerm]:
    """
    Extract core pricing concepts from sections mentioning interest / fees.

    We look for:
    - Lines with "Applicable Margin" / "Applicable Rate"
    - Lines with "commitment fee" / "unused fee"
    - Any nearby percentages, plus benchmark hints and floors.
    """
    results: List[PricingTerm] = []

    for sec in sections:
        sec_text_lower = sec.text.lower()
        title_lower = sec.title.lower()
        if not any(k in sec_text_lower or k in title_lower for k in PRICING_SECTION_KEYWORDS):
            continue

        # Split section into sentences-ish chunks
        chunks = re.split(r"(?<=[.;])\s+", sec.text)
        for chunk in chunks:
            chunk_lower = chunk.lower()
            if not any(word in chunk_lower for word in ["applicable margin", "applicable rate", "commitment fee", "unused fee", "floor"]):
                continue

            pct_matches = [m.group(0).strip() for m in PCT_RE.finditer(chunk)]
            if not pct_matches:
                continue

            benchmark_match = BENCHMARK_HINT_RE.search(chunk)
            benchmark_hint = benchmark_match.group(0).strip() if benchmark_match else None

            # Heuristically classify this line
            name = "Pricing Term"
            if "applicable margin" in chunk_lower:
                name = "Applicable Margin"
            elif "applicable rate" in chunk_lower:
                name = "Applicable Rate"
            elif "commitment fee" in chunk_lower or "unused fee" in chunk_lower:
                name = "Commitment Fee Rate"

            floors: List[str] = []
            other_rates: List[str] = pct_matches.copy()
            if "floor" in chunk_lower:
                # crude: treat all % in a 'floor' clause as floors
                floors = pct_matches.copy()
            else:
                floors = []

            pt = PricingTerm(
                name=name,
                benchmark_hint=benchmark_hint,
                margins=pct_matches if "margin" in chunk_lower or "rate" in chunk_lower else [],
                floors=floors,
                other_rates=other_rates,
                location_hint=sec.id,
                context_snippet=_snippet(chunk),
            )
            results.append(pt)

    return results
