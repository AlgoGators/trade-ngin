# ======================================================================
#  credit_doc_extractor.py  (single-file "multi-module" toolkit)
#
#  Purpose:
#    Offline extraction of key leveraged-finance provisions from legal PDFs:
#      • Debt baskets (incremental / starter / ratio-based / 1st / 2nd / unsecured lien / general / CF / other)
#      • MFN provisions
#      • Negative covenants & Restricted Payments (starter, builder, general RP, RP ratio basket, investments/JV/JDP, etc.)
#      • Financial covenants (maintenance + incurrence; total & lien-based leverage; gross/net; testing cadence; step-downs)
#      • EBITDA definitions (standard add-backs; pro forma; synergy/cost savings caps & look-forward)
#      • Pricing grid (leverage tiers; applicable margin; SOFR/benchmark & floor)
#      • Call protection; cash carve-outs; maturity; change of control
#      • Asset sale sweeps; ECF sweeps; mandatory prepayments
#
#  Design:
#    - No external web requests.
#    - Text extraction with pdfminer.six; optional OCR (pytesseract) for scanned PDFs.
#    - Heading-aware segmentation; configurable ontology of regex patterns.
#    - Windowed context (neighbor blocks) for clauses that spill across headings.
#    - Structured JSON output (+ confidence) and optional sections dump for QA.
#
#  CLI:
#    python credit_doc_extractor.py input.pdf --out results.json --min_conf 0.55
#    python credit_doc_extractor.py input.pdf --ocr
#    python credit_doc_extractor.py input.pdf --dump-sections sections.txt
#    python credit_doc_extractor.py input.pdf --ontology my_patterns.yaml   # optional, if PyYAML installed
#
#  Dependencies:
#    pip install pdfminer.six
#    # Optional OCR:
#    pip install pytesseract pdf2image pillow
#    # Optional external ontology:
#    pip install pyyaml
#
#  Note:
#    This code favors readability and extensibility. The default ontology covers
#    your requested classes, but you’ll likely iterate patterns to your house style.
# ======================================================================

from __future__ import annotations

import json
import re
from dataclasses import dataclass, asdict
from pathlib import Path
from typing import Dict, List, Optional, Tuple, Iterable, Any, Callable


@dataclass
class Block:
    """
    A logical block of text (typically under a heading).

    Attributes
    ----------
    heading : Optional[str]
        Heading line that precedes the block (e.g., "SECTION 6.01 Restricted Payments").
    text : str
        Body text for this block.
    idx : int
        Block index in document order.
    page_hint : Optional[int]
        Reserved for future layout-aware extraction; unused here.
    """
    heading: Optional[str]
    text: str
    idx: int
    page_hint: Optional[int] = None


@dataclass
class Match:
    """
    A provision match extracted from a block or window of blocks.

    Attributes
    ----------
    provision : str
        Canonical provision name (e.g., "Debt_Basket_Incremental").
    section_ref : Optional[str]
        Closest heading reference for reviewer navigation.
    block_idxs : List[int]
        One or more block indices that contributed to the snippet (for windowed captures).
    text : str
        Extracted snippet (sentence/paragraph region).
    amounts : List[str]
        Dollar amounts captured (e.g., "$100,000,000").
    ratios : List[str]
        Ratio tokens (e.g., "3.00x").
    percents : List[str]
        Percent tokens (e.g., "50%").
    dates : List[str]
        Date-like tokens (e.g., "July 1, 2029", "2029-07-01", "7/1/2029").
    years_terms : List[str]
        Tenor expressions (e.g., "5 years", "sixty (60) months").
    confidence : float
        Heuristic in [0, 1].
    """
    provision: str
    section_ref: Optional[str]
    block_idxs: List[int]
    text: str
    amounts: List[str]
    ratios: List[str]
    percents: List[str]
    dates: List[str]
    years_terms: List[str]
    confidence: float

    def to_dict(self) -> Dict[str, Any]:
        return asdict(self)


@dataclass
class ExtractionResult:
    """
    Container for an extraction run.
    """
    file: str
    total_blocks: int
    total_matches: int
    retained: int
    results: List[Match]

    def to_json(self, indent: int = 2) -> str:
        payload = {
            "file": self.file,
            "total_blocks": self.total_blocks,
            "total_matches": self.total_matches,
            "retained": self.retained,
            "results": [m.to_dict() for m in self.results],
        }
        return json.dumps(payload, indent=indent)

# Common token extractors you can reuse in downstream analytics.
AMOUNT_REGEX = re.compile(r"\$\s?\d[\d,]*(?:\.\d+)?")
RATIO_REGEX = re.compile(r"\b\d+(?:\.\d+)?x\b", re.IGNORECASE)
PCT_REGEX   = re.compile(r"\b\d+(?:\.\d+)?\s?%\b")

# Dates: textual and numeric (very permissive)
MONTHS = r"January|February|March|April|May|June|July|August|September|October|November|December"
DATE_REGEX = re.compile(
    rf"\b(?:\d{{1,2}}/\d{{1,2}}/\d{{2,4}}|(?:{MONTHS})\s+\d{{1,2}},\s+\d{{4}}|\d{{4}}-\d{{2}}-\d{{2}})\b",
    re.IGNORECASE
)

# Tenor / maturity terms
YEARS_TERMS_REGEX = re.compile(
    r"\b(?:(\d{1,2})\s+years?|(\d{1,3})\s+months?|(\w+)\s*\(\d{1,3}\)\s+months?)\b",
    re.IGNORECASE
)

# Heading detection for segmentation
HEADING_PATTERNS: List[re.Pattern] = [
    re.compile(r"^\s*ARTICLE\s+[IVXLC]+\b.*", re.IGNORECASE),
    re.compile(r"^\s*SECTION\s+\d+(?:\.\d+)*\b.*", re.IGNORECASE),
    re.compile(r"^\s*\d+(?:\.\d+)+\s+.+$"),                  # 1.1, 2.3.4 etc.
    re.compile(r"^\s*[A-Z][A-Z \-/]{4,}\s*$"),               # SHOUTY headings
    re.compile(r"^\s*Pricing\s+Grid\b.*", re.IGNORECASE),    # explicit grid headings
]

# Line-level noise removal during normalization
HEADER_FOOTER_NOISE = re.compile(
    r"^\s*(Page\s+\d+\s+of\s+\d+|Execution\s+Version|Confidential|Draft\s+for\s+Discussion)\s*$",
    re.IGNORECASE
)

from typing import List

from pdfminer.high_level import extract_text

def extract_pdf_text(pdf_path: Path) -> str:
    """
    Extract text via pdfminer.six (no layout coordinates to keep deps minimal).
    """
    return extract_text(str(pdf_path))


def extract_pdf_text_with_ocr(pdf_path: Path) -> str:
    """
    OCR fallback for scanned PDFs (requires pytesseract/pdf2image/Pillow + Tesseract binary).
    """
    try:
        from pdf2image import convert_from_path
        from PIL import Image
        import pytesseract
    except Exception as exc:  # pragma: no cover
        raise RuntimeError(
            "OCR requested but pytesseract/pdf2image/Pillow are not available. "
            "Install them and ensure Tesseract is installed."
        ) from exc

    pages = convert_from_path(str(pdf_path))
    texts: List[str] = []
    for page_img in pages:
        gray = page_img.convert("L")
        texts.append(pytesseract.image_to_string(gray))
    return "\n\n".join(texts)


# === file: nlp/normalize.py =============================================
def normalize_text(doc_text: str) -> str:
    """
    Normalize extraction quirks:
      1) Drop header/footer boilerplate.
      2) Fix hyphenation across line breaks.
      3) Collapse excessive blank lines.
    """
    kept: List[str] = []
    for ln in doc_text.splitlines():
        if HEADER_FOOTER_NOISE.match(ln or ""):
            continue
        kept.append(ln)
    text = "\n".join(kept)
    text = re.sub(r"(\w)-\n(\s*)(\w)", r"\1\3", text)   # "addi-\n tional" -> "additional"
    text = re.sub(r"\n{3,}", "\n\n", text)              # collapse 3+ blank lines
    return text


# === file: nlp/segment.py ===============================================
def is_heading_line(line: str) -> bool:
    """Heuristic heading detector."""
    return any(p.search(line) for p in HEADING_PATTERNS)


def split_into_blocks(doc_text: str) -> List[Block]:
    """
    Heading-aware segmentation:
    - Flush a block whenever we see a new heading-like line.
    - Drop micro-fragments (<= 5 words).
    """
    lines = doc_text.splitlines()
    blocks: List[Block] = []
    buf: List[str] = []
    current_heading: Optional[str] = None
    idx = 0

    def flush():
        nonlocal buf, current_heading, idx
        if buf:
            body = "\n".join(buf).strip()
            if len(body.split()) > 5:
                blocks.append(Block(heading=current_heading, text=body, idx=idx))
                idx += 1
            buf = []

    for ln in lines:
        if is_heading_line(ln):
            flush()
            current_heading = ln.strip()
        else:
            buf.append(ln)
    flush()
    return blocks

Expanded provision ontology for leveraged-finance docs.

Each provision entry can define:
  - include:  list of compiled regex patterns that indicate a positive hit
  - boost:    list of compiled regex patterns that increase confidence when present
  - window:   int number of neighbor blocks to include around the matching block
              (captures cross-heading grids, step-down tables, etc.)
"""

def rx(p: str, flags: int = re.IGNORECASE | re.DOTALL) -> re.Pattern:
    """Helper to compile concise regex strings."""

# === file: nlp/score.py ==================================================
def score_snippet(snippet: str, heading: Optional[str], boost_patterns: List[re.Pattern]) -> float:
    """
    Simple heuristic:
      + baseline for length
      + boosts for keyword hits
      + soft bump for presence of tokens (amounts/ratios/%)
    """
    base = 0.5 + min(len(snippet) / 2200.0, 0.2)
    for pat in boost_patterns:
        hits = len(pat.findall(snippet))
        base += min(hits * 0.08, 0.32)
    # Token presence bump
    if AMOUNT_REGEX.search(snippet):
        base += 0.04
    if RATIO_REGEX.search(snippet):
        base += 0.04
    if PCT_REGEX.search(snippet):
        base += 0.04
    return max(0.0, min(base, 1.0))


# === file: nlp/match.py ==================================================
def window_blocks(blocks: List[Block], center_idx: int, span: int) -> Tuple[List[int], str]:
    """
    Build a window of neighboring blocks around center_idx.

    Parameters
    ----------
    blocks : List[Block]
    center_idx : int
    span : int

    Returns
    -------
    (indices, text)
        indices: list of block indices contributing to the window
        text: concatenated text (with headings inline for context)
    """
    start = max(0, center_idx - span)
    end = min(len(blocks), center_idx + span + 1)
    idxs = list(range(start, end))
    texts: List[str] = []
    for i in idxs:
        h = blocks[i].heading or ""
        t = blocks[i].text
        # Include heading inline to help regex that look for "Pricing Grid" etc.
        if h:
            texts.append(f"{h}\n{t}")
        else:
            texts.append(t)
    return idxs, "\n\n".join(texts)


def regex_search_in_text(
    provision: str,
    include_patterns: List[re.Pattern],
    boost_patterns: List[re.Pattern],
    text: str,
    section_ref: Optional[str],
    idxs: List[int],
) -> Iterable[Match]:
    """
    Run include regex over a given text chunk and yield matches.
    """
    for inc in include_patterns:
        for m in inc.finditer(text):
            snippet = m.group(0).strip()
            amounts = AMOUNT_REGEX.findall(snippet)
            ratios  = RATIO_REGEX.findall(snippet)
            perc    = PCT_REGEX.findall(snippet)
            dates   = DATE_REGEX.findall(snippet)
            years   = YEARS_TERMS_REGEX.findall(snippet)
            # Flatten tenor tuple matches to a readable token list
            years_terms = []
            for y in years:
                tok = " ".join([s for s in y if s])
                if tok:
                    years_terms.append(tok)

            conf = round(score_snippet(snippet, section_ref, boost_patterns), 3)
            yield Match(
                provision=provision,
                section_ref=section_ref,
                block_idxs=idxs,
                text=snippet,
                amounts=amounts,
                ratios=ratios,
                percents=perc,
                dates=[d[0] if isinstance(d, tuple) else d for d in dates] if dates else [],
                years_terms=years_terms,
                confidence=conf,
            )


def run_extraction(blocks: List[Block], ontology: Dict[str, Dict[str, Any]]) -> List[Match]:
    """
    Execute ontology-driven extraction using windowed context.
    """
    raw: List[Match] = []
    for i, blk in enumerate(blocks):
        for prov, spec in ontology.items():
            include: List[re.Pattern] = spec.get("include", [])
            boost:   List[re.Pattern] = spec.get("boost", [])
            span:    int = int(spec.get("window", 0))

            idxs, win_text = window_blocks(blocks, i, span)
            section_ref = blk.heading
            raw.extend(regex_search_in_text(prov, include, boost, win_text, section_ref, idxs))
    # Deduplicate by (provision, first 140 chars)
    seen: set[Tuple[str, str]] = set()
    uniq: List[Match] = []
    for m in raw:
        key = (m.provision, m.text[:140])
        if key in seen:
            continue
        seen.add(key)
        uniq.append(m)
    return uniq


# === file: io/ontology_loader.py ========================================
def maybe_load_external_ontology(path: Optional[str]) -> Optional[Dict[str, Dict[str, Any]]]:
    """
    Optionally load ontology from a YAML file if provided and PyYAML is installed.
    The YAML should map provision -> { include: [regex...], boost: [regex...], window: int }
    Regex strings will be compiled with IGNORECASE|DOTALL.

    Example YAML snippet:
      Debt_Basket_Incremental:
        include:
          - "(?i)(Incremental\\s+Facilities?).*"
        boost:
          - "(?i)(First\\s+Lien|Second\\s+Lien)"
        window: 1
    """
    if not path:
        return None
    p = Path(path)
    if not p.exists():
        raise FileNotFoundError(f"Ontology YAML not found: {p}")
    try:
        import yaml  # type: ignore
    except Exception as exc:  # pragma: no cover
        raise RuntimeError("PyYAML is required to load an external ontology (pip install pyyaml).") from exc

    data = yaml.safe_load(p.read_text(encoding="utf-8"))
    compiled: Dict[str, Dict[str, Any]] = {}
    for prov, spec in data.items():
        inc_raw = spec.get("include", []) or []
        boo_raw = spec.get("boost", []) or []
        win_raw = spec.get("window", 0)
        compiled[prov] = {
            "include": [rx(s) for s in inc_raw],
            "boost":   [rx(s) for s in boo_raw],
            "window":  int(win_raw),
        }
    return compiled

import argparse
from textwrap import dedent

def build_arg_parser() -> argparse.ArgumentParser:
    p = argparse.ArgumentParser(
        prog="credit_doc_extractor",
        formatter_class=argparse.RawTextHelpFormatter,
        description=dedent(
            """\
            Leveraged-finance provision extractor (offline).

            Examples:
              python credit_doc_extractor.py input.pdf --out results.json
              python credit_doc_extractor.py input.pdf --ocr --dump-sections sections.txt
              python credit_doc_extractor.py input.pdf --ontology my_patterns.yaml
            """
        ),
    )
    p.add_argument("pdf", type=str, help="Path to input PDF")
    p.add_argument("--out", type=str, default="", help="JSON output path (stdout if omitted)")
    p.add_argument("--min_conf", type=float, default=0.55, help="Minimum confidence to retain")
    p.add_argument("--ocr", action="store_true", help="Use OCR for scanned PDFs (pytesseract required)")
    p.add_argument("--dump-sections", type=str, default="", help="Write segmented sections to this file")
    p.add_argument("--ontology", type=str, default="", help="Path to external YAML ontology (optional)")
    return p


def write_sections_dump(blocks: List[Block], out_path: Path) -> None:
    lines: List[str] = []
    for b in blocks:
        lines.append(f"--- Block #{b.idx} ---")
        lines.append(f"HEADING: {b.heading or '(None)'}")
        lines.append(b.text)
        lines.append("")
    Path(out_path).write_text("\n".join(lines), encoding="utf-8")


def main() -> None:
    args = build_arg_parser().parse_args()
    pdf_path = Path(args.pdf)
    if not pdf_path.exists():
        raise FileNotFoundError(f"PDF not found: {pdf_path}")

    # 1) Text extraction
    if args.ocr:
        raw = extract_pdf_text_with_ocr(pdf_path)
    else:
        raw = extract_pdf_text(pdf_path)

    # 2) Normalize + segment
    norm = normalize_text(raw)
    blocks = split_into_blocks(norm)

    if args.dump_sections:
        write_sections_dump(blocks, Path(args.dump_sections))

    # 3) Ontology selection (external overrides default)
    external = maybe_load_external_ontology(args.ontology) if args.ontology else None
    ontology = external if external else ONTOLOGY

    # 4) Run extraction, threshold, package
    matches = run_extraction(blocks, ontology)
    kept = [m for m in matches if m.confidence >= args.min_conf]
    result = ExtractionResult(
        file=str(pdf_path),
        total_blocks=len(blocks),
        total_matches=len(matches),
        retained=len(kept),
        results=kept,
    )

    # 5) Emit JSON
    payload = result.to_json(indent=2)
    if args.out:
        Path(args.out).write_text(payload, encoding="utf-8")
    else:
        print(payload)


# === file: __main__ guard ===============================================
if __name__ == "__main__":
    main()
