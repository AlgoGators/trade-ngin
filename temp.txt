#!/usr/bin/env python3
"""
summarize_credit_extractions.py

Purpose
-------
Summarize and structure the large JSON output produced by credit_doc_extractor.py.
Outputs:
  1) Console summary of counts by provision
  2) Markdown report grouping provisions with top-N highlights
  3) Per-provision CSVs under an output folder
  4) Optional single "all_matches.csv" for spreadsheet pivots

Usage
-----
python summarize_credit_extractions.py extracted.json --out-dir reports --top-n 3 --min-conf 0.65 --write-all-csv

Design Notes
------------
- No external requests. Pure Python stdlib (plus optional pandas if you enable --write-all-csv-fast).
- Works directly with the schema produced by credit_doc_extractor.py:
    {
      "file": "...",
      "total_blocks": int,
      "total_matches": int,
      "retained": int,
      "results": [
        {
          "provision": str,
          "section_ref": str | null,
          "block_idxs": [int, ...],
          "text": str,
          "amounts": [str, ...],
          "ratios": [str, ...],
          "percents": [str, ...],
          "dates": [str, ...],
          "years_terms": [str, ...],
          "confidence": float
        }, ...
      ]
    }

Author: (you)
"""

from __future__ import annotations

import argparse
import csv
import json
from collections import defaultdict
from dataclasses import dataclass
from pathlib import Path
from typing import Any, Dict, Iterable, List, Mapping, MutableMapping, Optional, Sequence, Tuple


# -----------------------------
# Data structures
# -----------------------------

@dataclass(frozen=True)
class Match:
    provision: str
    section_ref: Optional[str]
    block_idxs: List[int]
    text: str
    amounts: List[str]
    ratios: List[str]
    percents: List[str]
    dates: List[str]
    years_terms: List[str]
    confidence: float


@dataclass(frozen=True)
class Extraction:
    file: str
    total_blocks: int
    total_matches: int
    retained: int
    results: List[Match]


# -----------------------------
# Load / filter helpers
# -----------------------------

def load_extraction(path: Path) -> Extraction:
    """Load the extractor JSON into typed objects."""
    raw = json.loads(path.read_text(encoding="utf-8"))
    results = [
        Match(
            provision=m["provision"],
            section_ref=m.get("section_ref"),
            block_idxs=list(m.get("block_idxs", [])),
            text=m.get("text", ""),
            amounts=list(m.get("amounts", [])),
            ratios=list(m.get("ratios", [])),
            percents=list(m.get("percents", [])),
            dates=list(m.get("dates", [])),
            years_terms=list(m.get("years_terms", [])),
            confidence=float(m.get("confidence", 0.0)),
        )
        for m in raw.get("results", [])
    ]
    return Extraction(
        file=str(raw.get("file", "")),
        total_blocks=int(raw.get("total_blocks", 0)),
        total_matches=int(raw.get("total_matches", 0)),
        retained=int(raw.get("retained", 0)),
        results=results,
    )


def filter_by_confidence(matches: Sequence[Match], min_conf: float) -> List[Match]:
    """Keep matches >= min_conf."""
    return [m for m in matches if m.confidence >= min_conf]


def group_by_provision(matches: Sequence[Match]) -> Dict[str, List[Match]]:
    """Group matches by provision key."""
    groups: Dict[str, List[Match]] = defaultdict(list)
    for m in matches:
        groups[m.provision].append(m)
    # sort each group by confidence desc, then by snippet length (shorter first)
    for k in groups:
        groups[k].sort(key=lambda x: (-x.confidence, len(x.text)))
    return groups


# -----------------------------
# Summaries
# -----------------------------

def summarize_counts(groups: Mapping[str, Sequence[Match]]) -> List[Tuple[str, int, float]]:
    """
    Returns a list of (provision, count, avg_confidence) sorted by count desc.
    """
    rows: List[Tuple[str, int, float]] = []
    for prov, items in groups.items():
        n = len(items)
        avg = sum(m.confidence for m in items) / max(1, n)
        rows.append((prov, n, round(avg, 3)))
    rows.sort(key=lambda r: (-r[1], r[0]))
    return rows


def extract_key_values(m: Match) -> Dict[str, str]:
    """
    Pull a compact set of key value tokens to display alongside the snippet.
    """
    def join_or_dash(items: Sequence[str], maxn: int = 6) -> str:
        if not items:
            return "—"
        cut = items[:maxn]
        s = ", ".join(cut)
        if len(items) > maxn:
            s += " …"
        return s

    return {
        "Amounts": join_or_dash(m.amounts),
        "Ratios": join_or_dash(m.ratios),
        "Percents": join_or_dash(m.percents),
        "Dates": join_or_dash(m.dates),
        "Tenor": join_or_dash(m.years_terms),
    }


# -----------------------------
# Markdown report
# -----------------------------

def write_markdown_report(
    out_md: Path,
    extraction: Extraction,
    groups: Mapping[str, Sequence[Match]],
    top_n: int = 3,
) -> None:
    """
    Write a concise Markdown report with top-N matches per provision.
    """
    lines: List[str] = []
    lines.append(f"# Credit Provisions Summary")
    lines.append("")
    lines.append(f"**Source file:** `{extraction.file}`")
    lines.append(f"**Blocks:** {extraction.total_blocks} | **Raw matches:** {extraction.total_matches} | **Retained:** {extraction.retained}")
    lines.append("")
    lines.append("## Counts by Provision")
    lines.append("")
    lines.append("| Provision | Matches | Avg Confidence |")
    lines.append("|---|---:|---:|")
    for prov, cnt, avg in summarize_counts(groups):
        lines.append(f"| `{prov}` | {cnt} | {avg:.3f} |")
    lines.append("")

    # Detailed sections
    for prov, items in groups.items():
        lines.append(f"## {prov}")
        for i, m in enumerate(items[:top_n]):
            kv = extract_key_values(m)
            lines.append(f"**Top {i+1} • conf={m.confidence:.3f} • section:** {m.section_ref or '(none)'}  ")
            lines.append(f"- Amounts: {kv['Amounts']}")
            lines.append(f"- Ratios: {kv['Ratios']}")
            lines.append(f"- Percents: {kv['Percents']}")
            lines.append(f"- Dates: {kv['Dates']}")
            lines.append(f"- Tenor: {kv['Tenor']}")
            lines.append("")
            # Show a trimmed snippet (avoid flooding)
            snippet = m.text.strip()
            snippet = snippet.replace("\n", " ").strip()
            if len(snippet) > 800:
                snippet = snippet[:800] + " …"
            lines.append(f"> {snippet}")
            lines.append("")
        lines.append("---")
        lines.append("")

    out_md.write_text("\n".join(lines), encoding="utf-8")


# -----------------------------
# CSV exports
# -----------------------------

CSV_HEADER = [
    "provision", "confidence", "section_ref", "block_idxs",
    "amounts", "ratios", "percents", "dates", "years_terms", "text",
]

def serialize_list(items: Sequence[Any]) -> str:
    """Serialize lists for CSV (compact)."""
    if not items:
        return ""
    return "; ".join(str(x) for x in items)


def write_per_provision_csvs(groups: Mapping[str, Sequence[Match]], out_dir: Path) -> None:
    """
    Write one CSV per provision for easy inspection in Excel.
    """
    out_dir.mkdir(parents=True, exist_ok=True)
    for prov, items in groups.items():
        safe = prov.replace("/", "_").replace(" ", "_")
        out_csv = out_dir / f"{safe}.csv"
        with out_csv.open("w", newline="", encoding="utf-8") as f:
            w = csv.writer(f)
            w.writerow(CSV_HEADER)
            for m in items:
                w.writerow([
                    m.provision, f"{m.confidence:.3f}", m.section_ref or "",
                    ",".join(str(i) for i in m.block_idxs),
                    serialize_list(m.amounts),
                    serialize_list(m.ratios),
                    serialize_list(m.percents),
                    serialize_list(m.dates),
                    serialize_list(m.years_terms),
                    m.text.replace("\n", " ").strip(),
                ])


def write_all_csv(matches: Sequence[Match], out_csv: Path) -> None:
    """
    Write a single wide CSV for all matches (slower but no pandas dependency).
    """
    out_csv.parent.mkdir(parents=True, exist_ok=True)
    with out_csv.open("w", newline="", encoding="utf-8") as f:
        w = csv.writer(f)
        w.writerow(CSV_HEADER)
        for m in matches:
            w.writerow([
                m.provision, f"{m.confidence:.3f}", m.section_ref or "",
                ",".join(str(i) for i in m.block_idxs),
                serialize_list(m.amounts),
                serialize_list(m.ratios),
                serialize_list(m.percents),
                serialize_list(m.dates),
                serialize_list(m.years_terms),
                m.text.replace("\n", " ").strip(),
            ])


# -----------------------------
# CLI
# -----------------------------

def build_arg_parser() -> argparse.ArgumentParser:
    p = argparse.ArgumentParser(
        prog="summarize_credit_extractions",
        description="Turn extractor JSON into a concise report and CSVs.",
    )
    p.add_argument("json", type=str, help="Path to extractor JSON")
    p.add_argument("--out-dir", type=str, default="summary_out", help="Output directory for report/CSVs")
    p.add_argument("--top-n", type=int, default=3, help="Top N matches per provision in the Markdown report")
    p.add_argument("--min-conf", type=float, default=0.60, help="Minimum confidence to include in summaries/CSVs")
    p.add_argument("--write-all-csv", action="store_true", help="Also write a single all_matches.csv")
    return p


def main() -> None:
    args = build_arg_parser().parse_args()
    json_path = Path(args.json)
    out_dir = Path(args.out_dir)
    out_dir.mkdir(parents=True, exist_ok=True)

    extraction = load_extraction(json_path)
    # Filter and group
    kept = filter_by_confidence(extraction.results, args.min_conf)
    groups = group_by_provision(kept)

    # Console summary
    print(f"File: {extraction.file}")
    print(f"Blocks: {extraction.total_blocks} | Raw matches: {extraction.total_matches} | Retained (>= {args.min_conf}): {len(kept)}")
    print("\nCounts by provision (retained):")
    for prov, cnt, avg in summarize_counts(groups):
        print(f"  {prov:35s}  {cnt:4d}  avg_conf={avg:.3f}")

    # Markdown report
    out_md = out_dir / "report.md"
    write_markdown_report(out_md, extraction, groups, top_n=args.top_n)
    print(f"\nMarkdown report: {out_md}")

    # Per-provision CSVs
    csv_dir = out_dir / "provisions"
    write_per_provision_csvs(groups, csv_dir)
    print(f"Per-provision CSVs: {csv_dir}")

    # Optional single CSV
    if args.write_all_csv:
        out_csv = out_dir / "all_matches.csv"
        write_all_csv(kept, out_csv)
        print(f"All-matches CSV: {out_csv}")


if __name__ == "__main__":
    main()
