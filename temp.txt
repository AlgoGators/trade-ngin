"""
Extract defined terms from definitions article(s).

This version is more robust to ugly PDF line breaks:

- It does NOT rely on a huge multi-line regex.
- It scans line-by-line for patterns like:
    "Consolidated EBITDA" means ...
    “Consolidated EBITDA” shall mean ...
- Once it finds a term line, it appends subsequent lines until:
    * It hits another line that looks like a new definition, or
    * It hits a strong break (blank line / section-like line).

We still highlight "Consolidated EBITDA" (or similar) as the EBITDA definition.
"""

from __future__ import annotations

import re
from typing import List, Optional

from models import Section, Definition


# Matches the *start* of a definition on a single line, e.g.:
#   "Consolidated EBITDA" means ...
#   “Consolidated EBITDA” shall mean ...
DEF_START_RE = re.compile(
    r"""
    ^\s*
    [\"“']                         # opening quote
    (?P<term>[^\"”']+?)            # term text
    [\"”']                         # closing quote
    \s+
    (?P<verb>means|shall\s+mean)   # "means" or "shall mean"
    \s+
    (?P<rest>.+?)                  # rest of the line (start of definition)
    $
    """,
    re.IGNORECASE | re.VERBOSE,
)


def _looks_like_definition_start(line: str) -> bool:
    """
    Heuristic: does a line look like the start of a new definition?

    Parameters
    ----------
    line : str
        A single line of text.

    Returns
    -------
    bool
        True if the line looks like "Term" means/shall mean ...
    """
    return DEF_START_RE.match(line) is not None


def _extract_definitions_from_text(text: str, location_hint: str) -> List[Definition]:
    """
    Extract definitions from the given raw text by scanning line-by-line.

    Parameters
    ----------
    text : str
        Text that may contain multiple definitions.
    location_hint : str
        Hint about where this text came from (e.g. section id).

    Returns
    -------
    List[Definition]
        Parsed definitions from the text.
    """
    lines = text.split("\n")
    definitions: List[Definition] = []

    idx = 0
    n = len(lines)

    while idx < n:
        line = lines[idx].rstrip()

        start_match = DEF_START_RE.match(line)
        if not start_match:
            idx += 1
            continue

        # We found a term line
        term = start_match.group("term").strip()
        definition_parts: List[str] = [start_match.group("rest").strip()]

        idx += 1

        # Accumulate subsequent lines that appear to be part of the same definition
        while idx < n:
            next_line = lines[idx].rstrip()

            # Stop if the next line looks like a new definition
            if _looks_like_definition_start(next_line):
                break

            # Stop if the line is empty and we already have some content
            if not next_line.strip() and definition_parts:
                break

            # Otherwise, treat as continuation of current definition
            # (common for long legal definitions)
            definition_parts.append(next_line.strip())
            idx += 1

        full_definition = " ".join(part for part in definition_parts if part)

        definitions.append(
            Definition(
                term=term,
                definition=full_definition,
                location_hint=location_hint,
            )
        )

    return definitions


def find_definition_sections(sections: List[Section]) -> List[Section]:
    """
    Identify sections that likely contain the main definitions article(s).

    Heuristics (in order):
    - Section titles containing the word "definition".
    - Section "Section 1.01" (very common).
    - If nothing matches, we return an empty list and the caller can
      decide to scan *all* sections.

    Parameters
    ----------
    sections : List[Section]
        All parsed sections.

    Returns
    -------
    List[Section]
        Candidate definition sections.
    """
    candidates: List[Section] = [
        s for s in sections if "definition" in s.title.lower()
    ]
    if candidates:
        return candidates

    for s in sections:
        if s.id.lower().startswith("section 1.01"):
            return [s]

    return []


def extract_definitions_from_section(section: Section) -> List[Definition]:
    """
    Extract defined terms within a given section using the line-based scanner.

    Parameters
    ----------
    section : Section
        A section that may contain defined terms.

    Returns
    -------
    List[Definition]
        List of Definition objects from that section.
    """
    return _extract_definitions_from_text(section.text, location_hint=section.id)


def extract_all_definitions(sections: List[Section]) -> List[Definition]:
    """
    Extract all definitions from the agreement.

    Strategy:
    - First, try the definition sections (if we can identify them).
    - If that yields very few definitions (e.g., 0 or 1), we fall back
      to scanning *all* sections, since some docs scatter definitions
      across multiple places.

    Parameters
    ----------
    sections : List[Section]
        All parsed sections.

    Returns
    -------
    List[Definition]
        Aggregated list of definitions.
    """
    # Pass 1: use obvious definition sections
    def_sections = find_definition_sections(sections)
    all_defs: List[Definition] = []
    for sec in def_sections:
        all_defs.extend(extract_definitions_from_section(sec))

    # If we basically got nothing, fall back to scanning everything
    if len(all_defs) <= 1:
        all_defs = []
        for sec in sections:
            all_defs.extend(extract_definitions_from_section(sec))

    return all_defs


def find_ebitda_definition(definitions: List[Definition]) -> Optional[Definition]:
    """
    Identify the EBITDA-style definition, preferring "Consolidated EBITDA".

    Parameters
    ----------
    definitions : List[Definition]
        All parsed definitions.

    Returns
    -------
    Optional[Definition]
        The most relevant EBITDA definition, if found.
    """
    ebitda_like_defs = [d for d in definitions if "ebitda" in d.term.lower()]

    if not ebitda_like_defs:
        return None

    for d in ebitda_like_defs:
        if "consolidated ebitda" in d.term.lower():
            return d

    return ebitda_like_defs[0]

"""
Extract covenant baskets from key sections (Indebtedness, Restricted Payments).

This module is purely regex / rule-based and designed to be robust to
messy section titles:

- We search both section titles AND body text for keywords.
- We pick the section with the strongest signal (most keyword hits).
- Within each section, we split into clauses ((a), (b), etc.) and
  extract numeric features: dollar amounts, percentages, ratios,
  "greater of X and Y" patterns.
"""

from __future__ import annotations

import re
from typing import List, Optional, Tuple

from models import Section, BasketInfo


# ---------------------------- Regex patterns -----------------------------

CLAUSE_SPLIT_RE = re.compile(
    r"\n\s*\(([a-z])\)\s",  # newline + "(a) ", "(b) ", etc.
    re.IGNORECASE,
)

DOLLAR_AMOUNT_RE = re.compile(
    r"\$[\s]*[\d,]+(?:\.\d+)?(?:\s*(?:million|billion|thousand))?",
    re.IGNORECASE,
)

PERCENTAGE_RE = re.compile(
    r"\b\d+(?:\.\d+)?\s*%",
    re.IGNORECASE,
)

RATIO_RE = re.compile(
    r"\b\d+(?:\.\d+)?\s*(?:to|:|-)\s*\d+(?:\.\d+)?\b",
    re.IGNORECASE,
)

GREATER_OF_RE = re.compile(
    r"greater\s+of\s+(?P<x>.+?)\s+and\s+(?P<y>.+?)(?:[.;,]|$)",
    re.IGNORECASE | re.DOTALL,
)


# ---------------------------- Core helpers -------------------------------

def split_section_into_clauses(section_text: str) -> List[Tuple[Optional[str], str]]:
    """
    Split a section's text into clause-level items based on "(a)", "(b)" etc.

    Parameters
    ----------
    section_text : str
        Full text of a covenant section.

    Returns
    -------
    List[Tuple[Optional[str], str]]
        List of (clause_label, clause_text) tuples.

        - clause_label: e.g., "(a)", "(b)", or None if no labels detected.
        - clause_text: full text of the clause including the label.
    """
    text = section_text
    matches = list(CLAUSE_SPLIT_RE.finditer(text))

    # No clause markers; treat entire section as a single clause
    if not matches:
        return [(None, text.strip())]

    clauses: List[Tuple[Optional[str], str]] = []

    for idx, match in enumerate(matches):
        start_idx = match.start()
        label_char = match.group(1).lower()
        label = f"({label_char})"

        end_idx = matches[idx + 1].start() if idx + 1 < len(matches) else len(text)
        clause_text = text[start_idx:end_idx].strip()

        clauses.append((label, clause_text))

    return clauses


def extract_numeric_features(
    text: str,
) -> Tuple[List[str], List[str], List[str], List[Tuple[str, str]]]:
    """
    Extract numeric patterns from a clause of text.

    Returns
    -------
    Tuple[List[str], List[str], List[str], List[Tuple[str, str]]]
        (dollar_amounts, percentages, ratios, greater_of_expressions)
    """
    dollar_amounts = [m.group(0).strip() for m in DOLLAR_AMOUNT_RE.finditer(text)]
    percentages = [m.group(0).strip() for m in PERCENTAGE_RE.finditer(text)]
    ratios = [m.group(0).strip() for m in RATIO_RE.finditer(text)]

    greater_of_expressions: List[Tuple[str, str]] = []
    for m in GREATER_OF_RE.finditer(text):
        x_raw = m.group("x").strip()
        y_raw = m.group("y").strip()
        greater_of_expressions.append((x_raw, y_raw))

    return dollar_amounts, percentages, ratios, greater_of_expressions


def extract_baskets_from_section(section: Section) -> List[BasketInfo]:
    """
    Extract basket-level information from a covenant section.

    Parameters
    ----------
    section : Section
        The covenant section (e.g., Indebtedness, Restricted Payments).

    Returns
    -------
    List[BasketInfo]
        List of BasketInfo instances for each clause.
    """
    clauses = split_section_into_clauses(section.text)
    baskets: List[BasketInfo] = []

    for clause_label, clause_text in clauses:
        dollar_amounts, percentages, ratios, greater_of_expressions = extract_numeric_features(
            clause_text
        )

        baskets.append(
            BasketInfo(
                section_id=section.id,
                section_title=section.title,
                clause_label=clause_label,
                raw_text=clause_text,
                dollar_amounts=dollar_amounts,
                percentages=percentages,
                ratios=ratios,
                greater_of_expressions=greater_of_expressions,
            )
        )

    return baskets


# ---------------------- Section selection heuristics ---------------------

def _score_section_for_keywords(
    section: Section,
    keywords: List[str],
    weight_title: float = 2.0,
) -> float:
    """
    Compute a simple relevance score for a section given a list of keywords.

    - Each occurrence of a keyword in the title is given `weight_title`.
    - Each occurrence in the body text is given weight 1.0.

    Parameters
    ----------
    section : Section
        Section to score.
    keywords : List[str]
        Lowercase keywords to look for.
    weight_title : float
        Weight multiplier for matches in the title.

    Returns
    -------
    float
        Relevance score (higher is better).
    """
    score = 0.0
    title_lower = section.title.lower()
    text_lower = section.text.lower()

    for kw in keywords:
        score += weight_title * title_lower.count(kw)
        score += text_lower.count(kw)

    return score


def _find_best_section(
    sections: List[Section],
    keywords: List[str],
    min_score: float = 1.0,
) -> Optional[Section]:
    """
    Find the section with the highest keyword-based relevance score.

    Parameters
    ----------
    sections : List[Section]
        All parsed sections.
    keywords : List[str]
        Lowercase keywords for this covenant type.
    min_score : float
        Minimum score required to accept a match.

    Returns
    -------
    Optional[Section]
        The best matching section, or None if no section exceeds min_score.
    """
    best_section: Optional[Section] = None
    best_score: float = 0.0

    for sec in sections:
        score = _score_section_for_keywords(sec, keywords)
        if score > best_score:
            best_score = score
            best_section = sec

    if best_section is not None and best_score >= min_score:
        return best_section

    return None


def find_indebtedness_section(sections: List[Section]) -> Optional[Section]:
    """
    Locate the section that governs limitations on Indebtedness.

    We look for:
    - Titles and text containing words like "indebtedness", "debt",
      "incur indebtedness", etc.
    """
    keywords = [
        "indebtedness",
        "debt",
        "incur indebtedness",
        "incurrence of indebtedness",
    ]
    return _find_best_section(sections, keywords, min_score=1.0)


def find_restricted_payments_section(sections: List[Section]) -> Optional[Section]:
    """
    Locate the section that governs Restricted Payments.

    We look for:
    - Titles/text containing "restricted payments", "restricted payment",
      and also common phrasing about dividends / distributions.
    """
    keywords = [
        "restricted payments",
        "restricted payment",
        "dividends",
        "distributions",
        "stock repurchases",
    ]
    return _find_best_section(sections, keywords, min_score=1.0)
