from __future__ import annotations

from dataclasses import dataclass, asdict
from pathlib import Path
from typing import Any, Dict, List, Optional, Tuple


@dataclass
class Block:
    """
    Logical block/section of the PDF after heading-aware segmentation.
    """
    heading: Optional[str]
    text: str
    idx: int
    page_hint: Optional[int] = None


@dataclass
class Candidate:
    """
    A candidate snippet to feed to the LLM for a specific provision.
    """
    provision: str
    block_idxs: List[int]
    section_ref: Optional[str]
    snippet: str
    heuristic_conf: float


@dataclass
class LLMFieldSpec:
    """
    Field specification for a provision's schema (used for validation/coercion).
    """
    name: str
    type: str  # "str" | "float" | "int" | "bool" | "date" | "percent" | "ratio" | "currency" | "tenor"
    required: bool = False


@dataclass
class ProvisionSchema:
    """
    Strict schema for an LLM extraction of a provision.
    """
    provision: str
    fields: List[LLMFieldSpec]


@dataclass
class LLMExtraction:
    """
    Parsed/validated output from the LLM for one (provision, candidate).
    """
    provision: str
    section_ref: Optional[str]
    block_idxs: List[int]
    snippet: str
    values: Dict[str, Any]           # validated/coerced field values
    model_confidence: Optional[float]  # optional numeric confidence from LLM (0-1)
    raw_json: Dict[str, Any]

    def to_dict(self) -> Dict[str, Any]:
        d = {
            "provision": self.provision,
            "section_ref": self.section_ref,
            "block_idxs": self.block_idxs,
            "snippet": self.snippet,
            "values": self.values,
            "model_confidence": self.model_confidence,
            "raw_json": self.raw_json,
        }
        return d


@dataclass
class ExtractionBundle:
    """
    Container for a full-document run.
    """
    file: str
    blocks: int
    candidates: int
    extractions: List[LLMExtraction]

    def to_dict(self) -> Dict[str, Any]:
        return {
            "file": self.file,
            "blocks": self.blocks,
            "candidates": self.candidates,
            "extractions": [e.to_dict() for e in self.extractions],
        }

from __future__ import annotations

from pathlib import Path
from typing import List

from pdfminer.high_level import extract_text


def extract_pdf_text(path: Path) -> str:
    """
    Extract text from a text-based PDF using pdfminer.six.
    """
    return extract_text(str(path))


def extract_pdf_text_with_ocr(path: Path) -> str:
    """
    OCR fallback for scanned PDFs (requires pytesseract/pdf2image/Pillow + OS tesseract).
    """
    try:
        from pdf2image import convert_from_path
        from PIL import Image
        import pytesseract
    except Exception as exc:  # pragma: no cover
        raise RuntimeError(
            "OCR requested but pytesseract/pdf2image/Pillow are not available. "
            "Install them and ensure tesseract is installed."
        ) from exc

    pages = convert_from_path(str(path))
    texts: List[str] = []
    for img in pages:
        gray = img.convert("L")
        texts.append(pytesseract.image_to_string(gray))
    return "\n\n".join(texts)

from __future__ import annotations

import re
from typing import List

from core.regexes import HEADER_FOOTER_NOISE


def normalize_text(doc_text: str) -> str:
    """
    Normalize PDF quirks: strip boilerplate lines, fix hyphenation, collapse blank lines.
    """
    kept: List[str] = []
    for ln in doc_text.splitlines():
        if HEADER_FOOTER_NOISE.match(ln or ""):
            continue
        kept.append(ln)
    text = "\n".join(kept)
    text = re.sub(r"(\w)-\n(\s*)(\w)", r"\1\3", text)  # "addi-\n tional" -> "additional"
    text = re.sub(r"\n{3,}", "\n\n", text)
    return text

from __future__ import annotations

from typing import List, Optional

from core.regexes import HEADING_PATS
from core.types import Block


def _is_heading(line: str) -> bool:
    return any(p.search(line) for p in HEADING_PATS)


def split_into_blocks(text: str) -> List[Block]:
    """
    Heading-aware segmentation into Blocks.
    """
    lines = text.splitlines()
    blocks: List[Block] = []
    buf: List[str] = []
    heading: Optional[str] = None
    idx = 0

    def flush() -> None:
        nonlocal idx, buf, heading
        if buf:
            body = "\n".join(buf).strip()
            if len(body.split()) > 5:
                blocks.append(Block(heading=heading, text=body, idx=idx))
                idx += 1
            buf = []

    for ln in lines:
        if _is_heading(ln):
            flush()
            heading = ln.strip()
        else:
            buf.append(ln)
    flush()
    return blocks

from __future__ import annotations

import re
from typing import Dict, Iterable, List, Pattern, Tuple

from core.regexes import ANCHORS, AMOUNT_RE, RATIO_RE, PCT_RE
from core.types import Block, Candidate


def _compile_patterns(pats: Iterable[str]) -> List[Pattern[str]]:
    return [re.compile(p, re.IGNORECASE | re.DOTALL) for p in pats]


def mine_candidates(
    blocks: List[Block],
    max_per_provision: int = 6,
    window: int = 1,
) -> List[Candidate]:
    """
    Use lightweight anchors to propose candidate snippets per provision.
    We also add a small window of neighboring blocks to capture grids or split clauses.
    """
    out: List[Candidate] = []
    compiled: Dict[str, List[Pattern[str]]] = {k: _compile_patterns(v) for k, v in ANCHORS.items()}

    for i, blk in enumerate(blocks):
        context_idxs = list(range(max(0, i - window), min(len(blocks), i + window + 1)))
        context_text = "\n\n".join([(blocks[j].heading or "") + "\n" + blocks[j].text for j in context_idxs])

        for prov, pats in compiled.items():
            if any(p.search(context_text) for p in pats):
                # Basic heuristic confidence: presence of key tokens bumps score
                score = 0.5
                if AMOUNT_RE.search(context_text):
                    score += 0.08
                if RATIO_RE.search(context_text):
                    score += 0.08
                if PCT_RE.search(context_text):
                    score += 0.06
                # Cap
                score = min(1.0, score)
                out.append(
                    Candidate(
                        provision=prov,
                        block_idxs=context_idxs,
                        section_ref=blk.heading,
                        snippet=context_text[:5000],  # keep prompt payload bounded
                        heuristic_conf=round(score, 3),
                    )
                )

    # Keep top-N per provision by heuristic_conf, then prefer shorter snippets
    pruned: List[Candidate] = []
    by_prov: Dict[str, List[Candidate]] = {}
    for c in out:
        by_prov.setdefault(c.provision, []).append(c)
    for prov, items in by_prov.items():
        items.sort(key=lambda x: (-x.heuristic_conf, len(x.snippet)))
        pruned.extend(items[:max_per_provision])
    return pruned

from __future__ import annotations

import re
from typing import Any, Dict, Optional

from core.regexes import AMOUNT_RE, RATIO_RE, PCT_RE, DATE_RE, TENOR_RE


def _to_float_safe(s: str) -> Optional[float]:
    try:
        return float(s)
    except Exception:
        return None


def coerce_currency(s: str) -> Optional[float]:
    """
    Convert strings like "$100,000,000" or "100,000,000 USD" to float (raw units).
    """
    m = AMOUNT_RE.search(s)
    if not m:
        return None
    val = m.group(0)
    val = val.replace("$", "").replace(",", "").replace("USD", "").replace("usd", "").replace("dollars", "").strip()
    return _to_float_safe(val)


def coerce_ratio(s: str) -> Optional[float]:
    """
    Convert "3.50x" -> 3.5
    """
    m = RATIO_RE.search(s)
    if not m:
        return None
    return _to_float_safe(m.group(0).lower().replace("x", ""))


def coerce_percent(s: str) -> Optional[float]:
    """
    Convert "50%" -> 0.50 (fraction). If you prefer bps/percent, adjust here.
    """
    m = PCT_RE.search(s)
    if not m:
        return None
    pct = m.group(0).replace("%", "").strip()
    f = _to_float_safe(pct)
    return None if f is None else f / 100.0


def coerce_date(s: str) -> Optional[str]:
    """
    Return the matched date string as ISO-ish if possible, else the raw match.
    (To keep stdlib-only, we avoid heavy date parsing; you can plug in your preferred parser.)
    """
    m = DATE_RE.search(s)
    return m.group(0) if m else None


def coerce_int(s: str) -> Optional[int]:
    try:
        return int(s)
    except Exception:
        # try float -> int
        try:
            return int(float(s))
        except Exception:
            return None


def coerce_tenor_years(s: str) -> Optional[int]:
    """
    From phrases like "5 years" or "60 months" derive an integer years value.
    """
    m = TENOR_RE.search(s)
    if not m:
        return None
    years, months = m.groups()
    if years:
        return int(years)
    if months:
        return max(1, round(int(months) / 12))
    return None


def coerce_value_by_type(type_name: str, raw: Any) -> Optional[Any]:
    """
    Coerce raw LLM field values (which may be strings) to the expected type.
    """
    if raw is None:
        return None
    if type_name == "str":
        return str(raw)
    if type_name == "bool":
        if isinstance(raw, bool):
            return raw
        s = str(raw).strip().lower()
        if s in {"true", "yes", "y", "1"}:
            return True
        if s in {"false", "no", "n", "0"}:
            return False
        return None
    if type_name == "int":
        if isinstance(raw, (int,)):
            return int(raw)
        return coerce_int(str(raw))
    if type_name == "float":
        if isinstance(raw, (int, float)):
            return float(raw)
        try:
            return float(str(raw))
        except Exception:
            return None
    if type_name == "currency":
        return coerce_currency(str(raw))
    if type_name == "ratio":
        return coerce_ratio(str(raw))
    if type_name == "percent":
        return coerce_percent(str(raw))
    if type_name == "date":
        return coerce_date(str(raw))
    if type_name == "tenor":
        return coerce_tenor_years(str(raw))
    return None


def validate_and_coerce(schema_fields: Dict[str, str], data: Dict[str, Any]) -> Optional[Dict[str, Any]]:
    """
    Validate `data` keys exist and coerce each value according to schema_fields mapping.
    schema_fields: name -> type string
    """
    out: Dict[str, Any] = {}
    for name, tname in schema_fields.items():
        val = data.get(name, None)
        coerced = coerce_value_by_type(tname, val)
        if coerced is None:
            out[name] = None
        else:
            out[name] = coerced
    return out

SYSTEM_RULES = (
    "You are a precise legal extraction engine for leveraged-loan credit agreements. "
    "Return STRICT JSON ONLY, no prose. If a value does not appear in the provided text, return null. "
    "Do not invent. Extract exact numbers/ratios/percents/dates exactly as written, then normalize if obvious."
)

def schema_to_json_template(schema: ProvisionSchema) -> str:
    """
    Build a JSON skeleton (as text) the LLM must fill in.
    """
    fields = []
    for f in schema.fields:
        sample = "null"
        if f.type in {"str"}:
            sample = "\"\""
        elif f.type in {"bool"}:
            sample = "false"
        elif f.type in {"int", "float"}:
            sample = "0"
        elif f.type in {"percent", "ratio", "currency", "date", "tenor"}:
            sample = "null"
        fields.append(f"\"{f.name}\": {sample}")
    fields.append("\"model_confidence\": 0")
    return "{\n  " + ",\n  ".join(fields) + "\n}"


def build_prompt(
    provision: str,
    snippet: str,
    section_ref: str | None,
) -> str:
    """
    Construct a deterministic prompt with schema instructions and the snippet context.
    """
    schema = SCHEMAS[provision]
    template_json = schema_to_json_template(schema)
    field_descs = "\n".join([f"- {f.name} ({f.type}){' [required]' if f.required else ''}" for f in schema.fields])

    prompt = f"""
{SYSTEM_RULES}

Provision: {provision}
Section: {section_ref or '(none)'}
Schema fields:
{field_descs}

Return ONLY a JSON object with these keys. Use null for unknown.
Template (structure only; fill in values):
{template_json}

TEXT TO EXTRACT FROM (verbatim):
\"\"\" 
{snippet}
\"\"\"
"""
    return prompt.strip()

def safe_parse_llm_json(s: str) -> Dict[str, Any]:
    """
    Parse the model's JSON string, trimming whitespace and ignoring BOMs.
    """
    text = s.strip().lstrip("\ufeff")
    # If the model accidentally wraps JSON in code fences, strip them:
    if text.startswith("```"):
        text = text.strip("`")
        # remove potential json tag like ```json
        if text.lower().startswith("json"):
            text = text[4:]
        text = text.strip()
    return json.loads(text)

from __future__ import annotations

from typing import Any, Dict, List, Optional

from core.schemas import SCHEMAS
from core.types import Candidate, LLMExtraction
from llm.interface import call_rbc_llm, safe_parse_llm_json
from llm.prompts import build_prompt
from util.values import coerce_value_by_type


def _schema_field_type_map(provision: str) -> Dict[str, str]:
    return {f.name: f.type for f in SCHEMAS[provision].fields}


def extract_with_llm(candidate: Candidate) -> Optional[LLMExtraction]:
    """
    Single candidate → prompt → LLM → parse → validate/coerce.
    """
    prompt = build_prompt(candidate.provision, candidate.snippet, candidate.section_ref)
    raw = call_rbc_llm(prompt)  # <-- your integration point
    obj = safe_parse_llm_json(raw)

    # Pull optional model_confidence if present
    mc = obj.get("model_confidence", None)
    if isinstance(mc, (int, float)):
        model_conf = float(mc)
    else:
        model_conf = None

    # Validate/Coerce fields
    tmap = _schema_field_type_map(candidate.provision)
    values: Dict[str, Any] = {}
    for name, tname in tmap.items():
        values[name] = coerce_value_by_type(tname, obj.get(name, None))

    return LLMExtraction(
        provision=candidate.provision,
        section_ref=candidate.section_ref,
        block_idxs=candidate.block_idxs,
        snippet=candidate.snippet,
        values=values,
        model_confidence=model_conf,
        raw_json=obj,
    )


def rank_and_reduce(extractions: List[LLMExtraction]) -> List[LLMExtraction]:
    """
    If multiple candidates for the same provision exist, keep the 'best'.
    Heuristic: prefer explicit numeric coverage and highest (heuristic_conf + model_confidence).
    (We assume upstream candidate order already favors relevance.)
    """
    by_prov: Dict[str, List[LLMExtraction]] = {}
    for e in extractions:
        by_prov.setdefault(e.provision, []).append(e)

    final: List[LLMExtraction] = []
    for prov, items in by_prov.items():
        def score(e: LLMExtraction) -> float:
            has_number = any(
                isinstance(v, (int, float)) and v not in (None, 0)
                for v in e.values.values()
            )
            base = 0.5 + (0.2 if has_number else 0.0)
            mc = e.model_confidence if e.model_confidence is not None else 0.0
            return base + min(mc, 0.5)

        items.sort(key=lambda x: score(x), reverse=True)
        final.append(items[0])
    return final

from __future__ import annotations

from pathlib import Path
from typing import Iterable, List, Optional, Sequence, Set

from core.types import ExtractionBundle, LLMExtraction
from io.pdf_reader import extract_pdf_text, extract_pdf_text_with_ocr
from nlp.normalize import normalize_text
from nlp.segment import split_into_blocks
from nlp.candidates import mine_candidates
from llm.run import extract_with_llm, rank_and_reduce


def run_pipeline(
    pdf_path: Path,
    provisions_whitelist: Optional[Sequence[str]] = None,
    ocr: bool = False,
    max_candidates_per_provision: int = 6,
    window_blocks: int = 1,
) -> ExtractionBundle:
    """
    End-to-end pipeline: PDF -> blocks -> candidates -> LLM -> reconciled extractions.
    """
    if not pdf_path.exists():
        raise FileNotFoundError(pdf_path)

    raw = extract_pdf_text_with_ocr(pdf_path) if ocr else extract_pdf_text(pdf_path)
    norm = normalize_text(raw)
    blocks = split_into_blocks(norm)

    candidates = mine_candidates(blocks, max_per_provision=max_candidates_per_provision, window=window_blocks)
    if provisions_whitelist:
        allow: Set[str] = set(provisions_whitelist)
        candidates = [c for c in candidates if c.provision in allow]

    extractions: List[LLMExtraction] = []
    for cand in candidates:
        try:
            e = extract_with_llm(cand)
            if e:
                extractions.append(e)
        except Exception as exc:
            # Continue on LLM/parsing errors; you can log if desired.
            continue

    final = rank_and_reduce(extractions)
    return ExtractionBundle(
        file=str(pdf_path),
        blocks=len(blocks),
        candidates=len(candidates),
        extractions=final,
    )

from __future__ import annotations

import csv
from pathlib import Path
from typing import Dict, List

from core.types import ExtractionBundle, LLMExtraction


def write_json(bundle: ExtractionBundle, out_path: Path) -> None:
    out_path.write_text(__import__("json").dumps(bundle.to_dict(), indent=2), encoding="utf-8")


def write_markdown(bundle: ExtractionBundle, out_md: Path) -> None:
    lines: List[str] = []
    lines.append(f"# Credit Provisions (LLM-validated)")
    lines.append("")
    lines.append(f"**Source:** `{bundle.file}`")
    lines.append(f"**Blocks:** {bundle.blocks} | **Candidates:** {bundle.candidates} | **Final extractions:** {len(bundle.extractions)}")
    lines.append("")

    for e in bundle.extractions:
        lines.append(f"## {e.provision}")
        lines.append(f"**Section:** {e.section_ref or '(none)'}  ")
        if e.model_confidence is not None:
            lines.append(f"**Model confidence:** {e.model_confidence:.3f}  ")
        lines.append("")
        # Values table
        lines.append("| Field | Value |")
        lines.append("|---|---|")
        for k, v in e.values.items():
            lines.append(f"| `{k}` | {v if v is not None else '—'} |")
        lines.append("")
        # Snippet quote
        snippet = e.snippet.replace("\n", " ").strip()
        if len(snippet) > 800:
            snippet = snippet[:800] + " …"
        lines.append(f"> {snippet}")
        lines.append("")
        lines.append("---")
        lines.append("")
    out_md.write_text("\n".join(lines), encoding="utf-8")


def write_csv(bundle: ExtractionBundle, out_csv: Path) -> None:
    """
    Flat CSV: one row per final extraction with values flattened.
    """
    out_csv.parent.mkdir(parents=True, exist_ok=True)
    # Gather all field keys across provisions for a stable header
    field_keys: List[str] = sorted({k for e in bundle.extractions for k in e.values.keys()})
    header = ["provision", "section_ref", "model_confidence"] + field_keys + ["snippet"]

    with out_csv.open("w", newline="", encoding="utf-8") as f:
        w = csv.writer(f)
        w.writerow(header)
        for e in bundle.extractions:
            row = [
                e.provision,
                e.section_ref or "",
                f"{e.model_confidence:.3f}" if e.model_confidence is not None else "",
            ]
            for k in field_keys:
                v = e.values.get(k, None)
                row.append("" if v is None else v)
            row.append(e.snippet.replace("\n", " ").strip())
            w.writerow(row)

from __future__ import annotations

import argparse
from pathlib import Path
from typing import List, Optional

from core.schemas import SCHEMAS
from extractor.pipeline import run_pipeline
from reporting.summarize import write_csv, write_json, write_markdown


def _parse_list_arg(s: Optional[str]) -> Optional[List[str]]:
    if not s:
        return None
    return [x.strip() for x in s.split(",") if x.strip()]


def build_arg_parser() -> argparse.ArgumentParser:
    p = argparse.ArgumentParser(
        prog="credit_llm_extractor",
        description="LLM-assisted extraction of key leveraged-finance provisions from PDFs (offline except your LLM call).",
    )
    p.add_argument("pdf", type=str, help="Path to input PDF")
    p.add_argument("--ocr", action="store_true", help="Use OCR fallback (pytesseract/pdf2image/Pillow + tesseract)")
    p.add_argument("--provisions", type=str, default="", help="Comma-separated subset of provision keys to extract")
    p.add_argument("--window", type=int, default=1, help="Neighboring blocks to include around a candidate")
    p.add_argument("--max-candidates", type=int, default=6, help="Max candidates per provision")
    p.add_argument("--out-dir", type=str, default="llm_out", help="Output directory")
    return p


def main() -> None:
    args = build_arg_parser().parse_args()
    pdf_path = Path(args.pdf)
    out_dir = Path(args.out_dir)
    out_dir.mkdir(parents=True, exist_ok=True)

    provisions_whitelist = _parse_list_arg(args.provisions)
    if provisions_whitelist:
        unknown = [p for p in provisions_whitelist if p not in SCHEMAS]
        if unknown:
            raise ValueError(f"Unknown provision keys: {unknown}")

    bundle = run_pipeline(
        pdf_path=pdf_path,
        provisions_whitelist=provisions_whitelist,
        ocr=bool(args.ocr),
        max_candidates_per_provision=int(args.max_candidates),
        window_blocks=int(args.window),
    )

    write_json(bundle, out_dir / "extractions.json")
    write_markdown(bundle, out_dir / "report.md")
    write_csv(bundle, out_dir / "extractions.csv")

    print(f"Done. Outputs in: {out_dir.resolve()}")
    print("  - extractions.json")
    print("  - report.md")
    print("  - extractions.csv")


if __name__ == "__main__":
    main()
